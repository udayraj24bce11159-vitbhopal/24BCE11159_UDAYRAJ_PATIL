select * from worker;

select First_name, department, salary from worker where department='HR' or department='Admin'
order by salary asc;
select department, sum(salary)from worker group by department;
select First_name,department,joining_date from worker where salary>100000 order by joining_date asc;
select department, avg(salary) from worker group by department order by department asc;
select department, count(worker_id) from worker group by department;
select department, sum(salary)from worker group by department having sum(salary)<=300000;

    select distinct salary from worker w2
    where w1.department = w2.department
    order by salary desc
    limit 1 offset 1
