Q1.List the Employee Name, their Department Name, and the Location Name for all employees.

Ans: SELECT 
    E.employee_name,
    D.dept_name,
    D.location_id
FROM Employees E
JOIN Departments D
ON E.dept_id = D.dept_id;


Q2.Retrieve the Project Name, Client Name, and the Total Hours Worked on that project, but only for projects managed by the 'Consulting' department.

ans: SELECT 
    P.project_name,
    P.client_name,
    SUM(A.hours_worked) AS total_hours
FROM Projects P
JOIN Departments D
ON P.dept_id = D.dept_id
JOIN Assignments A
ON P.project_id = A.project_id
WHERE D.dept_name = 'Consulting'
GROUP BY P.project_name, P.client_name;


Q3.Find the names of Employees who have worked on a project with Client Name 'GlobalTech Solutions'. List the Employee Name and Project Name.

ANS: SELECT 
    E.employee_name,
    P.project_name
FROM Employees E
JOIN Assignments A
ON E.employee_id = A.employee_id
JOIN Projects P
ON A.project_id = P.project_id
WHERE P.client_name = 'GlobalTech Solutions';


Q4.List all Departments that have no Employees currently assigned to them.

ANS: SELECT 
    D.dept_name
FROM Departments D
LEFT JOIN Employees E
ON D.dept_id = E.dept_id
WHERE E.employee_id IS NULL;


Q5.Show the Manager Name, the Department Name they manage, and the total number of direct reports they have.

ANS: SELECT 
    M.employee_name AS manager_name,
    D.dept_name,
    COUNT(E.employee_id) AS total_reports
FROM Employees E
JOIN Employees M
ON E.manager_id = M.employee_id
JOIN Departments D
ON M.dept_id = D.dept_id
GROUP BY M.employee_name, D.dept_name;


Q6.Calculate the Average Hours Worked per Employee across all projects. List the Employee Name and the average hours.

ANS: SELECT 
    E.employee_name,
    AVG(A.hours_worked) AS avg_hours
FROM Employees E
JOIN Assignments A
ON E.employee_id = A.employee_id
GROUP BY E.employee_name;


Q7.Find the Department Name(s) that have an average salary above the company-wide average salary.

ANS : SELECT 
    D.dept_name
FROM Employees E
JOIN Departments D
ON E.dept_id = D.dept_id
GROUP BY D.dept_name
HAVING AVG(E.salary) > (
    SELECT AVG(salary)
    FROM Employees
);


Q8.Identify the Client Name(s) whose projects have a total combined hours worked exceeding 500 hours.

ANS: SELECT 
    P.client_name,
    SUM(A.hours_worked) AS total_hours
FROM Projects P
JOIN Assignments A
ON P.project_id = A.project_id
GROUP BY P.client_name
HAVING SUM(A.hours_worked) > 500;


Q10.Calculate the total number of employees assigned to each Project, but only include projects that have more than 5 employees assigned

ANS: SELECT 
    P.project_name,
    COUNT(DISTINCT A.employee_id) AS total_employees
FROM Projects P
JOIN Assignments A
ON P.project_id = A.project_id
GROUP BY P.project_name
HAVING COUNT(DISTINCT A.employee_id) > 5;


Q13.Write a query to list the Employee Name and Project Name for any employee who has worked less than 10 hours on a project.

ANS: SELECT 
    E.employee_name,
    P.project_name
FROM Employees E
JOIN Assignments A
ON E.employee_id = A.employee_id
JOIN Projects P
ON A.project_id = P.project_id
WHERE A.hours_worked < 10;


Q14.Find the Project Name(s) that have no assigned employees.

ANS: SELECT 
    P.project_name
FROM Projects P
LEFT JOIN Assignments A
ON P.project_id = A.project_id
WHERE A.assignment_id IS NULL;


Q16.Assuming the Assignments table has a start_date column, find the Employee Name and their oldest project start date.

ANS: SELECT 
    E.employee_name,
    MIN(A.start_date) AS oldest_project_date
FROM Employees E
JOIN Assignments A
ON E.employee_id = A.employee_id
GROUP BY E.employee_name;


Q17.Write a MySQL query using the DATE_FORMAT function to list all projects that were active in the year 2023

ANS: SELECT 
    project_name
FROM Projects
WHERE DATE_FORMAT(start_date, '%Y') = '2023';


Q18.Use the MySQL-specific IFNULL function to list all Employee Names. If an employee has no manager, display "CEO" instead of NULL.

ANS: SELECT 
    employee_name,
    IFNULL(manager_id, 'CEO') AS manager
FROM Employees;


Q19.Find the top 3 Departments based on the total hours worked by all their employees across all projects.

ANS: SELECT 
    D.dept_name,
    SUM(A.hours_worked) AS total_hours
FROM Departments D
JOIN Employees E
ON D.dept_id = E.dept_id
JOIN Assignments A
ON E.employee_id = A.employee_id
GROUP BY D.dept_name
ORDER BY total_hours DESC
LIMIT 3;


Q21.
List the Employee Name and their Manager's Name, even if the employee has no manager, or if a potential manager has no employees reporting to them.

ANS:SELECT 
    E.employee_name AS employee,
    M.employee_name AS manager
FROM Employees E
LEFT JOIN Employees M
ON E.manager_id = M.employee_id;


Q22.Write a query using a Common Table Expression (CTE) to find the Project Name and the number of unique employees assigned to it.

ANS: WITH ProjectEmployees AS (
    SELECT 
        P.project_name,
        COUNT(DISTINCT A.employee_id) AS unique_employees
    FROM Projects P
    LEFT JOIN Assignments A
    ON P.project_id = A.project_id
    GROUP BY P.project_name
)
SELECT * 
FROM ProjectEmployees;
