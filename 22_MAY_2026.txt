USE vitcampus;

DROP TABLE IF EXISTS HOD;
DROP TABLE IF EXISTS departments;

CREATE TABLE departments(
    did INT PRIMARY KEY,
    dname VARCHAR(25) NOT NULL UNIQUE
);

CREATE TABLE HOD(
    hid INT PRIMARY KEY,
    hname VARCHAR(20) NOT NULL UNIQUE
);

SELECT *
FROM departments
CROSS JOIN HOD
ORDER BY did ASC;

SELECT *
FROM departments
INNER JOIN HOD
ON departments.did = HOD.hid;

SELECT *
FROM departments
NATURAL JOIN HOD;

SELECT *
FROM departments
LEFT OUTER JOIN HOD
ON departments.did = HOD.hid
ORDER BY did ASC;

FROM departments
RIGHT OUTER JOIN HOD
ON departments.did = HOD.hid
